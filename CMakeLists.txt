cmake_minimum_required(VERSION 3.5)

project(EMT-show VERSION 0.1 LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


find_package(QT NAMES Qt6 Qt5 COMPONENTS Widgets Qt5SerialPort REQUIRED)
find_package(Qt${QT_VERSION_MAJOR} COMPONENTS Widgets SerialPort REQUIRED)
find_package(VTK REQUIRED)

include (${VTK_USE_FILE})
include_directories(${VTK_INCLUDE_DIRS}
                    ${PROJECT_SOURCE_DIR})


set(PROJECT_SOURCES
    main.cpp
    mainwindow.cpp
    ButtonUI.cpp
    TrackWork.cpp
    )

set(Heads
    mainwindow.h
    ButtonUI.h
    TrackWork.h
    )

set(UI
    mainwindow.ui
    picture.qrc
    )


if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(EMT-show
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )
else()
    add_executable(EMT-show
    ${PROJECT_SOURCES}
    ${Heads}
    ${UI})
endif()


target_link_libraries(EMT-show PRIVATE 
                      ${VTK_LIBRARIES}
                      Qt${QT_VERSION_MAJOR}::Core
                      Qt${QT_VERSION_MAJOR}::Gui
                      Qt${QT_VERSION_MAJOR}::Widgets
	      Qt${QT_VERSION_MAJOR}::SerialPort
                      )

if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(EMT-show)
endif()
